import pandas as pd import matplotlib.pyplot as plt # Dữ liệu về các tỷ phú Việt Nam (tài sản tính bằng tỷ USD) data = { 'Tỉ phú': [ 'Phạm Nhật Vượng', 'Nguyễn Thị Phương Thảo', 'Phạm Thu Hương', 'Trần Đình Long', 'Hồ Hùng Anh', 'Phạm Thúy Hằng', 'Ngô Chí Dũng', 'Nguyễn Đăng Quang' ], 'Tài sản trước (2024/2025)': [6.5, 2.8, 'Mới', 2.5, 2.0, 'Mới', 'Mới', 1.0], 'Tài sản hiện tại (2/2026)': [26.1, 4.1, 2.8, 2.7, 2.4, 2.0, 1.1, 1.1], 'Biến động': ['+19.6', '+1.3', '+2.8', '+0.2', '+0.4', '+2.0', '+1.1', '+0.1'] } # Tạo DataFrame df = pd.DataFrame(data) # Hiển thị bảng print(df) # Xử lý cột Biến động để vẽ biểu đồ (chuyển sang float, loại bỏ dấu '+') df['Biến động_num'] = df['Biến động'].str.replace('+', '').astype(float) # Vẽ biểu đồ cột plt.figure(figsize=(10, 6)) # Kích thước biểu đồ plt.bar(df['Tỉ phú'], df['Biến động_num'], color='skyblue') plt.xticks(rotation=45, ha='right') # Xoay nhãn để dễ đọc plt.xlabel('Tỉ phú') plt.ylabel('Biến động (tỷ USD)') plt.title('Biến động tài sản của các tỷ phú Việt Nam (2024/2025 đến 2/2026)') plt.tight_layout() # Tối ưu hóa layout plt.show() # Hiển thị biểu đồ
